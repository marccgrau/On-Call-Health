```python
@patch('app.api.endpoints.api_keys.APIKeyService')
def test_create_api_key_success(self, mock_service_class, client, mock_user):
    # Setup mock
    mock_service = Mock()
    mock_service_class.return_value = mock_service

    mock_api_key = Mock()
    mock_api_key.id = 1
    mock_api_key.name = "Test Key"
    mock_api_key.masked_key = "och_live_****abcd"
    mock_api_key.scope = "full_access"
    mock_api_key.created_at = datetime.now(timezone.utc)
    mock_api_key.expires_at = None

    mock_service.create_key.return_value = (mock_api_key, "och_live_fullkeyhere")

    # Make request
    response = client.post("/api/api-keys", json={"name": "Test Key"})

    # Verify
    assert response.status_code == 201
    data = response.json()
    assert data["name"] == "Test Key"
    assert data["key"] == "och_live_fullkeyhere"
```
